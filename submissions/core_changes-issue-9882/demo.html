Before:

Files to Modify

build-minified-css.mjs

// Current: custom regex minifier
// Change:  use csso with sourceMap:true

function minifyCss(css) {
  return minify(css, {
    filename: "easemotion.css",
    sourceMap: true,
  });
}

// Write .css and .map
const result = minifyCss(fullCss);
await writeFile("easemotion.min.css",
  result.css + "\n/*# sourceMappingURL=..." */
);
await writeFile("easemotion.min.css.map",
  JSON.stringify(result.map)
);
