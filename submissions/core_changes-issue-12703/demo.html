Usage

<!-- Padding all sides -->
<div class="ease-p-4">Content</div>

<!-- Horizontal + vertical padding -->
<div class="ease-px-6 ease-py-4">Content</div>

<!-- Margin top + auto horizontal -->
<div class="ease-mt-8 ease-mx-auto">Centered</div>

<!-- Individual sides -->
<div class="ease-mt-4 ease-mr-2 ease-mb-6 ease-ml-8">Content</div>
