3. Code Integration Guide

Toggle switches require simple semantic HTML markup. The styling is entirely self-contained within EaseMotion's variables and classes. 

<!-- Basic Accessible Switch -->
<label class="ease-toggle ease-toggle--md ease-toggle--primary">
  <input type="checkbox" class="ease-toggle-input">
  <span class="ease-toggle-track">
    <span class="ease-toggle-knob"></span>
  </span>
  <span class="ease-toggle-label">Toggle Options</span>
</label>
<!-- Add ease-toggle--bouncy modifier for spring feedback -->
<label class="ease-toggle ease-toggle--md ease-toggle--primary ease-toggle--bouncy">
  <input type="checkbox" class="ease-toggle-input">
  <span class="ease-toggle-track">
    <span class="ease-toggle-knob"></span>
  </span>
  <span class="ease-toggle-label">Bouncy Toggle</span>
</label>
<!-- Add ease-toggle--elastic modifier for stretch feedback -->
<label class="ease-toggle ease-toggle--md ease-toggle--primary ease-toggle--elastic">
  <input type="checkbox" class="ease-toggle-input">
  <span class="ease-toggle-track">
    <span class="ease-toggle-knob"></span>
  </span>
  <span class="ease-toggle-label">Fluid Elastic Toggle</span>
</label>
<!-- Add ease-toggle--flip modifier for 3D flip feedback -->
<label class="ease-toggle ease-toggle--md ease-toggle--primary ease-toggle--flip">
  <input type="checkbox" class="ease-toggle-input">
  <span class="ease-toggle-track">
    <span class="ease-toggle-knob"></span>
  </span>
  <span class="ease-toggle-label">3D Rotate Toggle</span>
</label>
<!-- Add ease-toggle--neon modifier for pulsing drop shadow -->
<label class="ease-toggle ease-toggle--md ease-toggle--primary ease-toggle--neon">
  <input type="checkbox" class="ease-toggle-input">
  <span class="ease-toggle-track">
    <span class="ease-toggle-knob"></span>
  </span>
  <span class="ease-toggle-label">Neon Glow Toggle</span>
</label>
<!-- Insert SVGs inside ease-toggle-knob with appropriate classes -->
<label class="ease-toggle ease-toggle--md ease-toggle--primary">
  <input type="checkbox" class="ease-toggle-input">
  <span class="ease-toggle-track">
    <span class="ease-toggle-knob">
      <!-- Off State Icon -->
      <svg viewBox="0 0 24 24" class="ease-toggle-icon-off">...</svg>
      <!-- On State Icon -->
      <svg viewBox="0 0 24 24" class="ease-toggle-icon-on">...</svg>
    </span>
  </span>
  <span class="ease-toggle-label">Icon Morph Toggle</span>
</label>
