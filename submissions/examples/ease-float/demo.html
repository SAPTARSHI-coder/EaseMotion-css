11 — Usage

Drop it in and go

Add one class. Compose freely. No configuration required.

HTML 
<!-- Default float -->
<img class="ease-float" src="hero.svg" />

<!-- Hover only -->
<div class="ease-float-hover">
  Hover me
</div>

<!-- Run once -->
<span class="ease-float-once">✦</span>

<!-- Stagger with delay -->
<div class="ease-float"
     style="animation-delay:0.5s">
  🌟
</div>
Composing 
<!-- High + slow -->
<div class="ease-float-high
       ease-float-slow">
  🌙
</div>

<!-- Subtle + fast -->
<div class="ease-float-subtle
       ease-float-fast">
  ⚡
</div>

<!-- Custom duration override -->
<div class="ease-float"
     style="animation-duration:2s">
  🚀
</div>
