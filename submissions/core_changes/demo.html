Usage

<button class="btn btn-primary btn-ripple" onclick="createRipple(event)">
  Click Me
</button>
.btn-ripple {
  position: relative;
  overflow: hidden;
}

@keyframes ripple {
  0% {
    transform: scale(0);
    opacity: 0.35;
  }
  100% {
    transform: scale(4);
    opacity: 0;
  }
}
